Step 02
Wire the robot circuit

Use the breadboard and MM jumper wires to connect the ESP8266 NodeMCU LoLin to the PIR sensor, servo, RTC, relay, buzzer, LED, and push button. The LoLin board labels 3.3V as 3U, ground as G, and 5V as VIN.

Step 03b
Copy & upload the code
5 min ▾

Create a new sketch in Arduino IDE, delete everything, and paste the code below. Replace the Blynk Template ID, Template Name, Auth Token, Wi-Fi name, and Wi-Fi password before uploading.

// PIR Light + Security Mode System with Blynk
// ESP8266 NodeMCU LoLin

#define BLYNK_TEMPLATE_ID "YOUR_TEMPLATE_ID"
#define BLYNK_TEMPLATE_NAME "Smart Lighting Security"
#define BLYNK_AUTH_TOKEN "YOUR_AUTH_TOKEN"

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>

char ssid[] = "YOUR_WIFI_NAME";
char pass[] = "YOUR_WIFI_PASSWORD";

#define PIR_PIN D5
#define WHITE_LED D0
#define RED_LED D8
#define BUZZER D7

bool securityMode = false; // false = normal, true = security

unsigned long lastMotionTime = 0;
const unsigned long lightHoldTime = 10000; // 10 seconds for white light

unsigned long lastSecurityMotionTime = 0;
const unsigned long alarmHoldTime = 5000; // 5 seconds after motion stops

unsigned long lastAlarmToggle = 0;
const unsigned long alarmInterval = 200;
bool alarmState = false;

bool notificationSent = false;

BLYNK_WRITE(V0) {
  securityMode = param.asInt();

  if (securityMode) {
    digitalWrite(WHITE_LED, LOW);
    stopAlarm();
    notificationSent = false;
    Serial.println("BLYNK: SECURITY MODE ON");
  } else {
    stopAlarm();
    Serial.println("BLYNK: NORMAL MODE ON");
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println();
  Serial.println("=== PIR LIGHT + SECURITY MODE SYSTEM ===");
  Serial.println("Blynk V0 controls Normal/Security mode");
  Serial.println("Type 'n' for NORMAL MODE");
  Serial.println("Type 's' for SECURITY MODE");

  pinMode(PIR_PIN, INPUT);
  pinMode(WHITE_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);
  pinMode(BUZZER, OUTPUT);

  digitalWrite(WHITE_LED, LOW);
  digitalWrite(RED_LED, LOW);
  digitalWrite(BUZZER, LOW);

  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);

  Serial.println("Waiting for PIR to calibrate...");
  for (int i = 60; i > 0; i--) {
    Serial.print("PIR ready in ");
    Serial.print(i);
    Serial.println(" seconds");
    delay(1000);
  }

  Serial.println("PIR ready!");
  Serial.println("Starting in NORMAL MODE");
  Blynk.virtualWrite(V0, 0);
}

void loop() {
  Blynk.run();
  checkSerialCommand();

  int motion = digitalRead(PIR_PIN);

  if (securityMode) {
    handleSecurityMode(motion);
  } else {
    handleNormalMode(motion);
  }

  delay(50);
}

void checkSerialCommand() {
  if (Serial.available() > 0) {
    char command = Serial.read();

    if (command == 's' || command == 'S') {
      securityMode = true;
      digitalWrite(WHITE_LED, LOW);
      stopAlarm();
      notificationSent = false;
      Blynk.virtualWrite(V0, 1);
      Serial.println("MODE CHANGED TO: SECURITY MODE");
    }

    if (command == 'n' || command == 'N') {
      securityMode = false;
      stopAlarm();
      Blynk.virtualWrite(V0, 0);
      Serial.println("MODE CHANGED TO: NORMAL MODE");
    }
  }
}

void handleNormalMode(int motion) {
  stopAlarm();
  notificationSent = false;

  if (motion == HIGH) {
    Serial.println("NORMAL MODE: Motion detected, white light ON");
    lastMotionTime = millis();
    digitalWrite(WHITE_LED, HIGH);
  } else {
    Serial.println("NORMAL MODE: No motion");

    if (millis() - lastMotionTime > lightHoldTime) {
      digitalWrite(WHITE_LED, LOW);
    }
  }
}

void handleSecurityMode(int motion) {
  digitalWrite(WHITE_LED, LOW);

  if (motion == HIGH) {
    Serial.println("SECURITY MODE: Motion detected, alarm ON");
    lastSecurityMotionTime = millis();

    if (!notificationSent) {
      Blynk.logEvent("intruder_alert", "WARNING! Intruder detected in your room!");
      notificationSent = true;
    }
  } else {
    Serial.println("SECURITY MODE: No motion");
  }

  // Keep alarm running for 5 seconds after last detected motion
  if (millis() - lastSecurityMotionTime <= alarmHoldTime) {
    flashAlarm();
  } else {
    stopAlarm();
    notificationSent = false;
  }
}

void flashAlarm() {
  if (millis() - lastAlarmToggle >= alarmInterval) {
    lastAlarmToggle = millis();

    alarmState = !alarmState;

    digitalWrite(RED_LED, alarmState);
    digitalWrite(BUZZER, alarmState);
  }
}

void stopAlarm() {
  digitalWrite(RED_LED, LOW);
  digitalWrite(BUZZER, LOW);
  alarmState = false;
}
📡 To change the timeout: edit TIMEOUT. The default is 300000 (5 minutes). For 10 minutes use 600000. For 30 minutes use 1800000.

Step 04
Test the system

Verify the servo scan, PIR sensor, relay, buzzer, red LED, RTC, and push button using the Serial Monitor before connecting a real light.

🧪
Step 04
Verify sensor & remote control
15 min ▾
1
Open Tools → Serial Monitor in Arduino IDE. Set baud rate to 115200.
2
You should see "RTC ready", then "PIR warming up...", and finally "System ready."
3
Check that the servo rotates the PIR sensor from 0° to 180°. Wave your hand in front of the PIR sensor; the yellow light/relay should turn ON in Normal Mode.
4
Press the push button to switch to Security Mode. Wave your hand again; the red LED, buzzer, and light should flash rapidly.
✅ If the servo scans, the relay turns the light on in Normal Mode, and the buzzer/LED flash in Security Mode, the robot is working correctly.
⚡
Step 05
Connect your light
15 min ▾
⚠️ Safety first: Make sure the light switch on your wall is turned OFF and the lamp is unplugged before doing this step. If you are unsure, ask someone with electrical experience to help.

The relay module has three screw terminals on one side: COM (Common), NO (Normally Open), and NC (Normally Closed). For our setup, use COM and NO.

1
Cut the live wire (usually brown or red) of your lamp cord. Do not cut the neutral or earth wire.
2
Connect one cut end to the COM terminal. Connect the other cut end to the NO terminal.
3
Plug the lamp back in and power the ESP8266 via USB. Wave your hand — the light should turn on. Wait 5 minutes without moving — it should turn off.
🎉 Setup complete! Your smart lighting robot can now scan the room, control the light, and trigger the intruder alarm in Security Mode.

Step 06
Setup Blynk
5 min

Install the free Blynk IoT app and create a new template for ESP8266 using Wi-Fi.

1
Create one virtual pin datastream: V0, type Integer, min 0, max 1. This controls Normal/Security mode.
2
Add a Switch widget and connect it to V0. OFF = Normal Mode. ON = Security Mode.
3
Create an event in Blynk called intruder_alert. This is used by the code to send the phone notification.
4
Copy the Template ID, Template Name, and Auth Token from Blynk and paste them into the code.
5
Upload the code. When the ESP8266 connects, the device should show as online in Blynk.
✅When the app switch is ON, motion will trigger the red LED, buzzer, and phone notification: WARNING! Intruder detected in your room!
